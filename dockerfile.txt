# استخدم صورة Node.js
FROM node:14

# تعيين المجلد العامل في الحاوية
WORKDIR /usr/src/app

# نسخ جميع ملفات المشروع إلى الحاوية
COPY . .

# تثبيت التبعيات من package.json
RUN npm install

# تعيين المنفذ الذي سيستمع عليه التطبيق (افتراضياً 3000)
EXPOSE 3000

# تحديد الأمر الذي سيقوم بتشغيله عند بدء الحاوية
CMD ["npm", "start"]
